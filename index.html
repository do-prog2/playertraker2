<script>

    let model;

    const video = document.getElementById('video-preview');

    const eventsDiv = document.getElementById('events');

    const status = document.getElementById('status');


    // Load the AI model on startup

    async function loadModel() {

        status.innerText = "Loading AI Vision Engine...";

        model = await cocoSsd.load();

        status.innerText = "AI Ready.";

    }

    loadModel();


    async function analyzeVideo() {

        const file = document.getElementById('videoUpload').files[0];

        if (!file || !model) return alert("Please upload a video and wait for the model to load.");


        video.src = URL.createObjectURL(file);

        video.style.display = "block";

        status.innerText = "Analyzing frames... (This may take a moment)";

        eventsDiv.innerHTML = "";


        // Wait for video metadata to load

        video.onloadeddata = async () => {

            let isPlayerOnField = false;

            const duration = video.duration;

            

            // Sample every 2 seconds of the video

            for (let i = 0; i < duration; i += 2) {

                video.currentTime = i;

                await new Promise(r => video.onseeked = r);

                

                const predictions = await model.detect(video);

                const foundPlayer = predictions.some(p => p.class === 'person');


                // Logic: If person detected, assume entry/exit (simplified for MVP)

                if (foundPlayer && !isPlayerOnField) {

                    logEvent(`[${formatTime(i)}] Player Entered Field`);

                    isPlayerOnField = true;

                } else if (!foundPlayer && isPlayerOnField) {

                    logEvent(`[${formatTime(i)}] Player Exited Field`);

                    isPlayerOnField = false;

                }

            }

            status.innerText = "Analysis Complete.";

        };

    }


    function logEvent(msg) {

        const div = document.createElement('div');

        div.className = 'event';

        div.innerText = msg;

        eventsDiv.appendChild(div);

    }


    function formatTime(seconds) {

        const m = Math.floor(seconds / 60);

        const s = Math.floor(seconds % 60);

        return `${m}:${s < 10 ? '0' : ''}${s}`;

    }
